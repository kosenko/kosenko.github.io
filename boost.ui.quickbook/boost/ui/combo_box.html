Description

Usage example: 

ui::combo_box combo_box = ui::combo_box(parent, "Initial text",
    { "Option 1", "Option 2", "Option 3" })
    .on_select_event([&](ui::index_event& e)
    {
        ui::log::info() << "Selected index: " << e.index()
                        << ", text: " << combo_box.text();
    });
